## 문자열 자르기

```c++
#include <iostream>
#include <string>
#include <vector>
#include <sstream>

string str = "java c c++ python";

istringstream istrstream(str);
string strBuffer;
vector<string> res;
res.clear();

while(getline(istrstream, strBuffer, ' ')) {
  res.push_back(strBuffer);
  cout << strBuffer << " ";
}

for(int i=0; i<res.size(); i++){
  cout << res[i] << '\n';
}
```
